def calculate_sum(n):
    # این تابع جمع اعداد از ۱ تا n را محاسبه می‌کند
    total = 0
    for i in range(1, n+1):
        total += i
    return total

def main():
    number = int(input("لطفاً یک عدد وارد کنید: "))
    result = calculate_sum(number)
    print(f"حاصل جمع اعداد از ۱ تا {number} برابر است با: {result}")

if __name__ == "__main__":
    main()
